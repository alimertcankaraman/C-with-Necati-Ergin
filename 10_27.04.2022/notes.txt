Copy Constructor
Hayatını bir başka nesnenin özelliğini alarak geliyor. (Aynı tür bir sınıf nesnesiden alabilir)

class MyClass
{

}

void func (MyClass);	//Call by value burda copy const çağrılıcak ilk değişken için

MyClass foo();

main()
{
	MyClass m1;
	MyClass m2(m1);
	MyClass m3{m1};
	MyClass m4 = m1;
	
	func(m1);	//copy constructor çağrılır
	
	MyClass m5 = foo();
}
